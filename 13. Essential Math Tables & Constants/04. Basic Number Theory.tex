\textbf{GCD / LCM:} \\[1mm]
$\gcd(a,b)$ by Euclid (iterative) \quad $\mathrm{lcm}(a,b)=\dfrac{|a b|}{\gcd(a,b)}$ \\

\textbf{Extended Euclid:} }Solve $ax+by=\gcd(a,b)$ for $x,y$ \\[1mm]

\textbf{Modular exponentiation:} \\[1mm]
Compute $a^b \bmod m$ in $O(\log b)$ by binary exponentiation \\

\textbf{Modular inverse:} \\[1mm]
If $\gcd(a,m)=1$, inverse exists. If $m$ prime: $a^{-1}\equiv a^{m-2}\pmod m$ (Fermat) \\

\textbf{Euler's totient:} \\[1mm]
$\varphi(n) = n\prod_{p\mid n}\left(1-\frac{1}{p}\right)$ \\

\textbf{Legendre formula (power of $p$ in $n!$):} \\[1mm]
$\nu_p(n!)=\sum_{i\ge1}\left\lfloor\frac{n}{p^i}\right\rfloor$ \\

\textbf{Wilson's theorem:} \\[1mm]
$p\text{ prime} \iff (p-1)!\equiv -1\pmod p$ \\
